def fibonacci(n: int) -> list[int]:
    """生成前 n 个斐波那契数"""
    seq = [0, 1]
    while len(seq) < n:
        seq.append(seq[-1] + seq[-2])
    return seq[:n]

if __name__ == "__main__":
    print(fibonacci(10))

Rust

fn is_palindrome(s: &str) -> bool {
    let t: String = s.chars().filter(|c| c.is_alphanumeric()).collect();
    t.chars().eq(t.chars().rev())
}

fn main() {
    println!("{}", is_palindrome("A man a plan a canal Panama"));
}

#include <stdio.h>

int sum(int arr[], int len) {
    int total = 0;
    for (int i = 0; i < len; i++) {
        total += arr[i];
    }
    return total;
}

int main(void) {
    int nums[] = {1, 2, 3, 4, 5};
    printf("sum = %d\n", sum(nums, 5));

SQL

SELECT p.title, c.name AS category, COUNT(t.id) AS tag_count
FROM posts p
JOIN categories c ON p.category_id = c.id
LEFT JOIN post_tags pt ON p.id = pt.post_id
LEFT JOIN tags t ON pt.tag_id = t.id
WHERE p.published = true
GROUP BY p.id, c.name
ORDER BY p.created_at DESC
LIMIT 10;
