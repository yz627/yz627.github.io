Tabs 多语言代码对比

下面展示 Hello World 在不同语言中的写法：

package main

import "fmt"

func main() {
    fmt.Println("Hello, World!")
}
def main():
    print("Hello, World!")

if __name__ == "__main__":
    main()
function main() {
    console.log("Hello, World!");
}

main();
fn main() {
    println!("Hello, World!");
}
